66  rw_lock(rw_semaphore& s_, bool write_): s(s_), write(write_) {
67  if(write) {
68  // Logic: write creates a persistent lock, preventing other writes from
69  // starting. They can still execute ++s.w to let themselves known but
70  // this enqueues them at the end of wq.wait. Write can only start when
71  // everyone is finished reading.
72  pers_lock = std::unique_lock<std::mutex>(s.m);
73  ++s.w;
74  s.wq.wait(pers_lock, [&] { return s.r == 0; });
75  ++s.mod_cnt;
76  } else {
77  // Logic: read creates a temporary lock for the purposes of rq.wait and
78  // s.r access. Read can only start when no one is writing or waiting to
79  // write.
80  std::unique_lock<std::mutex> temp_lock(s.m);
81  s.rq.wait(temp_lock, [&] { return s.w == 0; });
82  ++s.r;
83  }
84  }
85

107  /* Upgrade a read lock to a write lock, giving this thread priority over
108  * waiting write-only locks.
109  *
110  * Important: don't rely on any information about the resource obtained
111  * prior to upgrade()! If more readers simultaneously want to upgrade then
112  * they will have to wait for each other, and may get the lock in any
113  * order, changing the state visible to the other ones.
114  *
115  * However, a thread can check if its upgrade request was uninterrupted
116  * by other threads, if necessary, by comparing if s.mod_cnt raised by 1
117  * exactly.
118  *
119  * Returns whether an upgrade happened: the returned value is false if
120  * this was already a writer lock. This is useful for a conditional
121  * downgrade() later when given a lock of unknown type. */
122  bool upgrade() {
123  if(write)
124  return false;
125  // Atomically combined action of -read and +write
126  pers_lock = std::unique_lock<std::mutex>(s.m);
127  ++s.w;
128  --s.r;
129  write = true;
130  s.wq.wait(pers_lock, [&] { return s.r == 0; });
131  ++s.mod_cnt;
132  // It makes no sense for another waiting write to wake up now, that
133  // would defeat the purpose of the upgrade. But we need to notify the
134  // other threads that r might be zero as no one else would do that.
135  // Keeping pers_lock allows them to exit wait() but stay stuck at the
136  // mutex lock.
137  s.wq.notify_all();
138  return true;
139  }
140 
141  /* Upgrade if a condition is met. If another thread (with an upgraded read
142  * lock) breaks the condition in the time of obtaining the write lock,
143  * it's released again.
144  *
145  * Returns whether the upgrade was successful: when the return value is
146  * true, we (1) have a write lock and (2) are sure that cond() is true.
147  * Throws an exception if this was already a write lock, because it could
148  * not be distinguished whether the upgrade was not successful due to the
149  * condition or due to that. */
150  bool upgrade_if(std::function<bool()> cond) {
151  if(write)
152  throw std::logic_error("upgrade_if called on a write lock");
153  if(!cond())
154  return false;
155  upgrade();
156  if(!cond()) {
157  downgrade();
158  return false;
159  } else
160  return true;
161  }
162 
163  /* Downgrade a write lock to a read lock, allowing the read to finish
164  * before another writer obtains the lock. */
165  void downgrade() {
166  if(!write)
167  return;
168  // Atomically combined action of -write and +read.
169  --s.w;
170  ++s.r;
171  write = false;
172  pers_lock.unlock();
173  // Other waiting reads can proceed now, the write is done
174  s.rq.notify_all();
175  }
