372  template<class Rng = decltype(rng)>
373  void randomTrim(size_t newSize, Rng& rng = rng) {
374  internal::read_lock lock{smp};
375  if(!lock.upgrade_if([newSize,this]() -> bool { return size() > newSize; }))
376  return;
377  if(size() == 0)
378  return;
379  shuffle(rng);
380  /* A reference is needed for resize() to fill up new space if newSize >
381  * size(). (There's an overload without this argument but that relies on a
382  * default constructor which we don't require). This is prevented by the
383  * lock so we can pass any reference at hand. Front is guaranteed to exist
384  * after excluding an empty population above. */
385  auto& dummy = Base::front();
386  Base::resize(newSize, dummy);
387  }

405  NOINLINE void prune(bool (*test)(const Candidate<CBase>&),
406  size_t minSize = 0, Rng& rng = rng) {
407  internal::read_lock lock{smp};
408  if(!lock.upgrade_if([minSize,this]() -> bool { return size() > minSize; }))
409  return;
410  size_t sz = size();
411  for(size_t i = 0; i < sz; )
412  if(test(operator[](i)))
413  std::swap(Base::operator[](i), Base::operator[](--sz));
414  else
415  i++;

438  template<class Rng = decltype(rng)>
439  NOINLINE void prune(
440  bool (*test)(const Candidate<CBase>&, const Candidate<CBase>&),
441  size_t minSize = 0, bool randomize = true, Rng& rng = rng) {
442  internal::read_lock lock{smp};
443  if(!lock.upgrade_if([minSize,this]() -> bool { return size() > minSize; }))
444  return;
445  if(randomize)
446  shuffle(rng);
447  size_t sz = size();
448  for(size_t i = 0; i < sz - 1; i++) {
449  const Candidate<CBase>& op1 = operator[](i);
450  for(size_t j = i + 1; j < sz; ) {
451  if(test(op1, operator[](j)))
452  std::swap(Base::operator[](j), Base::operator[](--sz));
453  else
454  j++;
455  }
456  }
457  Base::erase(Base::begin() + sz, Base::end());
458  }

501  template<class Rng = decltype(rng)>
502  Ref randomSelect(size_t k, Rng& rng = rng) const {
503 #else
504  template<class Rng = decltype(rng), class Ret = Ref>
505  NOINLINE Ret randomSelect(size_t k, Rng& rng = rng) const {
506 #endif
507  internal::read_lock lock{smp};
508  size_t sz = size();
509  if(k >= sz)
510  return Ret{*this};
511  std::vector<size_t> idx(sz);
512  /* Fisher-Yates intentionally without initialization! */
513  for(size_t i = 0; i < k; i++) {
514  size_t d = rng() % (sz - i), j = i+d;
515  std::swap(idx[i], idx[j]);
516  idx[j] -= d;
517  idx[i] += i + d;
518  }
519  idx.resize(k);
520  Ret ret{k};
521  for(auto i : idx)
522  ret.add(operator[](i));
523  return ret;
524  }
