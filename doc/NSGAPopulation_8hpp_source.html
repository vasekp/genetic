158  NOINLINE void nsga_rate(bool parallel = false) {
159  std::vector<nsga_struct> vec{};
160  std::vector<nsga_struct*> ref{};
161  size_t sz = size();
162  /* vec: size() preallocated nsga_structs, each pointing to one Candidate
163  * ref: size() pointers to vec, originally 1:1 in order
164  * Other ways tried:
165  * - std::unique_ptr<nsga_struct>: unnecessary individual allocation
166  * - list embedded in a vector: slow due to special cases
167  * - validity flag: slow */
168  vec.reserve(sz);
169  ref.reserve(sz);
170  for(auto& tg : Base::as_vec()) {
171  vec.push_back({tg});
172  ref.push_back(&vec.back());
173  }
174  /* Calculate dominance counts and lists of dominated candidates */
175  #pragma omp parallel for if(parallel)
176  for(size_t i = 0; i < sz; i++) {
177  nsga_struct& op1 = vec[i];
178  for(nsga_struct& op2 : vec)
179  if(op1.rCand() << op2.rCand()) {
180  op1.dom.push_back(&op2);
181  #pragma omp atomic
182  op2.dom_cnt++;
183  }
184  }
185  /* ref contains the candidates with yet unassigned rank. When it becomes
186  * empty, we're done. */
187  size_t cur_rank = 0;
188  std::vector<nsga_struct> cur_front{};
189  auto first = ref.begin();
190  auto last = ref.end();
191  while(first != last) {
192  cur_front.clear();
193  for(auto it = first; it != last; )
194  if((*it)->dom_cnt == 0) {
195  cur_front.push_back(std::move(**it));
196  std::swap(*it, *(--last));
197  } else
198  it++;
199  /* Break arrows from members of cur_front and assign final rank to
200  * them */
201  for(auto& r : cur_front) {
202  for(auto pdom : r.dom)
203  pdom->dom_cnt--;
204  r.rRank() = cur_rank;
205  }
206  cur_rank++;
207  }
208  }

211  NOINLINE Ret NSGASelect_int(double bias, Rng& rng) {
212  internal::read_lock lock{smp};
213  size_t sz = size();
214  if(sz == 0)
215  throw std::out_of_range("NSGASelect(): Population is empty.");
216  internal::read_lock nsga_lock{nsga_smp};
217  if(smp.get_mod_cnt() != nsga_last_mod || bias != nsga_last_bias) {
218  nsga_lock.upgrade();
219  if(smp.get_mod_cnt() != nsga_last_mod)
220  nsga_rate(false);
221  nsga_probs.clear();
222  nsga_probs.reserve(sz);
223  for(auto& tg : Base::as_vec())
224  nsga_probs.push_back(1 / fun(tg.tag(), bias));
225  nsga_dist = std::discrete_distribution<size_t>
226  (nsga_probs.begin(), nsga_probs.end());
227  nsga_last_mod = smp.get_mod_cnt();
228  nsga_last_bias = bias;
229  }
230  size_t ix = nsga_dist(rng);
231  return operator[](ix);
232  }
