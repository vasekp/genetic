162  NOINLINE Ret rankSelect_exp(double bias, Rng& rng) {
163  internal::read_lock lock{smp};
164  double x = uniform(rng);
165  size_t sz = size();
166  if(sz == 0)
167  throw std::out_of_range("rankSelect(): Population is empty.");
168  ensure_sorted(lock);
169  // Bug in GCC: https://gcc.gnu.org/bugzilla/show_bug.cgi?id=63176
170  if(x == 1)
171  return Base::last();
172  else
173  return operator[]((int)(-log(1 - x + x*exp(-bias))/bias*sz));
174  }

176  template<class Ret, double (*fun)(double, double), class Rng>
177  NOINLINE Ret rankSelect_two(double bias, Rng& rng) {
178  internal::read_lock lock{smp};
179  size_t sz = size();
180  if(sz == 0)
181  throw std::out_of_range("rankSelect(): Population is empty.");
182  internal::read_lock sort_lock{sort_smp};
183  if(sz != rankSelect_last_sz || bias != rankSelect_last_bias) {
184  sort_lock.upgrade();
185  rankSelect_probs.clear();
186  rankSelect_probs.reserve(sz);
187  for(size_t i = 0; i < sz; i++)
188  rankSelect_probs.push_back(1 / fun((double)(i+1) / sz, bias));
189  rankSelect_dist = std::discrete_distribution<size_t>
190  (rankSelect_probs.begin(), rankSelect_probs.end());
191  rankSelect_last_sz = sz;
192  rankSelect_last_bias = bias;
193  }
194  ensure_sorted(lock);
195  return operator[](rankSelect_dist(rng));
196  }
